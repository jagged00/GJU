demonic_cripple_meridians = {
    category = interaction_category_prison
    common_interaction = yes
    interface_priority = 70
    icon = icon_combat

    desc = demonic_cripple_meridians_desc

    is_shown = {
        scope:actor = {
            is_cultivator = yes
        }
        scope:recipient = {
            is_imprisoned_by = scope:actor
            is_cultivator = yes
            NOT = { has_trait = crippled_meridians }
        }
    }

    is_valid = {
        scope:recipient = {
            is_imprisoned_by = scope:actor
            NOT = { has_trait = crippled_meridians }
        }
    }

    ai_will_do = {
        factor = 30
        modifier = { add = 120 has_trait = sadistic }
        modifier = { add = 100 has_trait = ruthless }
        modifier = { add = 80 has_relation_rival = scope:recipient }
        modifier = { add = 50 scope:recipient = { trait_rank.cultivation_realm >= 3 } }
    }

    on_accept = {
        scope:recipient = {
            remove_trait = qi_gathering
            remove_trait = foundation_establishment
            remove_trait = core_formation
            remove_trait = nascent_soul
            remove_trait = spirit_severing
            remove_trait = dao_seeking
            remove_trait = immortal_ascension

            remove_trait = poor_meridians
            remove_trait = normal_meridians
            remove_trait = good_meridians
            remove_trait = excellent_meridians
            remove_trait = heavenly_meridians
            add_trait = crippled_meridians

            add_stress = 120
            add_opinion = {
                target = scope:actor
                modifier = meridians_crippled_opinion
                opinion = -80
            }
        }
        scope:actor = {
            add_dread = 60
            add_prestige = 200
            add_tyranny = 30
            random_list = {
                80 = { }
                20 = { add_trait = wounded_3 }
            }
        }
    }
}
