# TEACH CULTIVATION – FINAL AI VERSION (CLOSE FAMILY FIRST + NO ERRORS)
#############################
teach_cultivation_interaction = {
	category = interaction_category_friendly
	interface_priority = 100
	icon = icon_combat
	
	desc = teach_cultivation_interaction_desc
	use_diplomatic_range = yes
	
	# Hyper-aggressive AI – checks very frequently
	ai_frequency = 150
	ai_targets = {
		ai_recipients = courtiers
		ai_recipients = children
		ai_recipients = siblings
		ai_recipients = parents
	}
	
	# FIXED: "this" error removed – correct vanilla syntax
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		
		scope:actor = {
			is_ai = yes
			is_alive = yes
			has_trait = cultivation_realm
		}
		scope:recipient = {
			is_alive = yes
			is_courtier_of = scope:actor
                        has_trait = active_meridians
			NOT = { has_trait = cultivation_realm }
		}
	}
	
	# AI WILL DO – CLOSE FAMILY IS GOD-TIER PRIORITY (includes poor meridians)
	ai_will_do = {
		base = 0
		
		# 1. CLOSE FAMILY IN COURT — TAUGHT FIRST, NO EXCEPTIONS
		modifier = {
			factor = 999999
			scope:recipient = {
				is_close_family_of = scope:actor
				is_courtier_of = scope:actor
				NOT = { has_trait = cultivation_realm }
			}
		}
		
		# 2. Extended family
		modifier = {
			factor = 10000
			scope:recipient = {
				is_extended_family_of = scope:actor
				is_courtier_of = scope:actor
				NOT = { has_trait = cultivation_realm }
			}
		}
		
		# 3. Non-family talented courtiers
		modifier = { factor = 2000  scope:recipient = { has_trait = heavenly_meridians } }
		modifier = { factor = 1200  scope:recipient = { has_trait = good_meridians } }
		modifier = { factor = 600   scope:recipient = { has_trait = normal_meridians } }
		
		# 4. Non-family poor meridians – only if literally nothing else exists
		modifier = {
			factor = 10
			scope:recipient = { has_trait = poor_meridians }
		}
		
		# 5. Never teach someone who already knows
		modifier = { factor = 0  scope:recipient = { has_trait = qi_gathering } }
		
		# 6. Slight opinion influence (won’t teach hated relatives)
		opinion_modifier = {
			who = scope:actor
			opinion_target = scope:recipient
			multiplier = 0.5
		}
	}
	
	ai_accept = { always = yes }
	auto_accept = yes
	
	on_accept = {
		scope:recipient = {
			add_trait = qi_gathering
			
			add_opinion = {
				target = scope:actor
				modifier = taught_me_cultivation_opinion
				years = 30
			}
		}
		scope:actor = {
			add_opinion = {
				target = scope:recipient
				modifier = taught_them_cultivation_opinion
				years = 30
			}
		}
	}
}

xianxia_extinguish_cultivator_family_interaction = {
    icon = extinguish_noble_family
    category = interaction_category_prison
    common_interaction = yes
    interface_priority = 65

    desc = xianxia_extinguish_cultivator_family_interaction_desc

    is_shown = {
        scope:actor = {
            has_trait = cultivation_realm
            is_landed = yes
            OR = {
                has_trait = core_formation
                has_trait = nascent_soul
                has_trait = spirit_severing
                has_trait = dao_seeking
                has_trait = immortal_ascension
            }
        }
        scope:recipient = {
            is_imprisoned_by = scope:actor
            has_trait = cultivation_realm
            is_landed = yes
        }
    }

    is_valid_showing_failures_only = {
        scope:recipient = {
            is_imprisoned_by = scope:actor
        }
    }

    auto_accept = { always = yes }

    ai_will_do = {
        base = 50
        modifier = { add = 100 has_trait = sadistic }
        modifier = { add = 80 has_trait = ruthless }
        modifier = { add = 50 has_relation_rival = scope:recipient }
        modifier = { add = 30 scope:recipient = { highest_held_title_tier >= tier_duchy } }
    }

    on_accept = {
        scope:recipient = {
            every_close_or_extended_family_member = {
                limit = {
                    is_alive = yes
                    has_trait = cultivation_realm
                    top_liege ?= scope:actor
                }
                death = {
                    death_reason = death_execution
                    killer = scope:actor
                }
            }

            death = {
                death_reason = death_execution
                killer = scope:actor
            }

            every_held_title = {
                destroy_title = this
            }
        }

        scope:actor = {
            add_dread = 80
            add_trait_xp = { trait = demonic_path value = 200 }
            add_character_modifier = { modifier = demonic_conquest_minor years = 5 }
        }
    }
}

cultivator_spar_interaction = {
	category = interaction_category_friendly
	interface_priority = 70
	icon = icon_combat
	desc = cultivator_spar_interaction_desc
	use_diplomatic_range = yes

	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = { has_trait = cultivation_realm }
		scope:recipient = { has_trait = cultivation_realm }
		OR = {
			scope:recipient = { has_relation_friend = scope:actor }
			scope:actor = { has_relation_friend = scope:recipient }
			scope:recipient = { opinion = { target = scope:actor value >= 40 } }
			scope:actor = { opinion = { target = scope:recipient value >= 40 } }
		}
	}

	ai_will_do = {
		base = 40
		modifier = { add = 40 has_trait = gregarious }
		modifier = { add = 30 has_trait = brave }
	}

	auto_accept = yes

	on_accept = {
		random_list = {
			75 = {
				scope:actor = { add_prestige = 50 }
				scope:recipient = { add_prestige = 50 }
			}
			20 = {
				scope:actor = { earn_cultivation_xp = yes }
				scope:recipient = { earn_cultivation_xp = yes }
			}
			5 = {
				scope:actor = { add_trait = wounded_1 }
				scope:recipient = { add_trait = wounded_1 }
			}
		}
	}
}
