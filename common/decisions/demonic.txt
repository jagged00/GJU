demonic_restore_meridians = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_refine_meridians.dds"
	}

	desc = demonic_restore_meridians_desc
	selection_tooltip = demonic_restore_meridians_selection_tooltip
	sort_order = 415
	decision_group_type = major

	is_shown = {
		has_trait = crippled_meridians
		OR = {
			gold >= 8000
			any_prisoner = {
				has_trait = heavenly_meridians
				is_imprisoned_by = root
			}
		}
	}

	is_valid = {
		OR = {
			gold >= 8000
			any_prisoner = {
				has_trait = heavenly_meridians
				is_imprisoned_by = root
			}
		}
	}

	ai_check_interval = 6
	ai_will_do = {
		base = 60
		modifier = { add = 220 has_trait = crippled_meridians }
		modifier = { add = 150 has_trait = ruthless }
		modifier = { add = 120 has_trait = sadistic }
		modifier = {
			add = 120
			any_prisoner = {
				has_trait = heavenly_meridians
				is_imprisoned_by = root
			}
		}
		modifier = { add = -120 has_trait = compassionate }
	}

	effect = {
		if = {
			limit = { gold >= 8000 }
			remove_gold = 8000
			random_list = {
				75 = {
					remove_trait = crippled_meridians
					add_trait = poor_meridians
					add_prestige = 350
					add_dread = 20
				}
				20 = {
					remove_trait = crippled_meridians
					add_trait = normal_meridians
					add_prestige = 600
					add_dread = 40
					add_stress = 40
				}
				5 = {
					add_trait = wounded_2
					add_stress = 120
				}
			}
		}
		else = {
			random_prisoner = {
				limit = {
					has_trait = heavenly_meridians
					is_imprisoned_by = root
				}
				save_scope_as = sacrifice
			}

			scope:sacrifice = {
				death = {
					death_reason = death_execution
					killer = root
				}
			}

			remove_trait = crippled_meridians
			remove_trait = poor_meridians
			remove_trait = normal_meridians
			remove_trait = good_meridians
			remove_trait = excellent_meridians
			remove_trait = heavenly_meridians

			random_list = {
				15 = {
					add_trait = heavenly_meridians
					add_dread = 150
					add_prestige = 900
					add_tyranny = 60
					add_piety = -180
				}
				85 = {
					add_trait = good_meridians
					add_dread = 70
					add_prestige = 450
					add_tyranny = 25
					add_stress = 80
					add_piety = -100
				}
			}
		}
	}

	cooldown = { years = 20 }
}
