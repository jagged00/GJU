# Yearly spontaneous jianghu brush with fate.
xianxia_depth.010 = {
	type = character_event
	title = xianxia_depth.010.t
	desc = xianxia_depth.010.desc
	theme = lifestyle_mystic
	left_portrait = { character = root animation = personality_bold }

	trigger = {
		is_adult = yes
		has_trait = cultivation_realm
	}

	option = {
		name = xianxia_depth.010.a
		ai_chance = { base = 45 }
		random_list = {
			50 = {
				add_character_modifier = { modifier = dao_heart_clarity years = 3 }
				add_piety = 75
			}
			25 = {
				create_breakthrough_artifact = { OWNER = root }
				add_prestige = 100
			}
			25 = {
				add_character_modifier = { modifier = jianghu_karmic_entanglement years = 3 }
				add_stress = 30
			}
		}
	}

	option = {
		name = xianxia_depth.010.b
		ai_chance = { base = 35 }
		add_character_modifier = { modifier = jianghu_fortuitous_encounter years = 2 }
		add_prestige = 75
		add_piety = 50
	}

	option = {
		name = xianxia_depth.010.c
		ai_chance = { base = 20 }
		add_stress = -20
		if = {
			limit = { has_character_modifier = heart_demon_afflicted }
			remove_character_modifier = heart_demon_afflicted
		}
	}
}

# Decision-driven deeper encounter arc (lightweight, one event only).
xianxia_depth.100 = {
	type = character_event
	title = xianxia_depth.100.t
	desc = xianxia_depth.100.desc
	theme = diplomacy_foreign_affairs
	left_portrait = { character = root animation = survey }

	trigger = {
		has_trait = cultivation_realm
		is_adult = yes
	}

	option = {
		name = xianxia_depth.100.a
		ai_chance = { base = 45 }
		random_list = {
			35 = {
				create_breakthrough_artifact = { OWNER = root }
				add_character_modifier = { modifier = dao_heart_clarity years = 2 }
			}
			45 = {
				if = {
					limit = { has_trait = righteous_path }
					add_trait_xp = { trait = righteous_path value = 80 }
				}
				add_piety = 120
				if = {
					limit = { has_trait = demonic_path }
					add_stress = 20
				}
			}
			20 = {
				add_trait = wounded_1
				add_stress = 40
			}
		}
	}

	option = {
		name = xianxia_depth.100.b
		ai_chance = { base = 35 }
		random_list = {
			40 = {
				if = {
					limit = { has_trait = demonic_path }
					add_trait_xp = { trait = demonic_path value = 80 }
				}
				add_character_modifier = { modifier = jianghu_karmic_entanglement years = 4 }
				add_prestige = 150
			}
			40 = {
				add_gold = 150
				add_character_modifier = { modifier = jianghu_fortuitous_encounter years = 2 }
			}
			20 = {
				add_trait = wounded_2
				add_stress = 60
			}
		}
	}

	option = {
		name = xianxia_depth.100.c
		ai_chance = { base = 20 }
		add_character_modifier = { modifier = dao_heart_clarity years = 1 }
		add_stress = -25
	}
}
