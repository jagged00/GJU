# Righteous traveler is targeted by a demonic marauder
xianxia_path.1001 = {
	type = character_event
	title = xianxia_path.1001.t
	desc = xianxia_path.1001.d
	theme = martial

	left_portrait = { character = root animation = personality_bold }
	right_portrait = { character = scope:ambusher animation = personality_evil }

	trigger = {
		is_available_travelling_adult = yes
		has_trait = righteous_path
		is_cultivator = yes
	}

	immediate = {
		random_living_character = {
			limit = {
				has_trait = demonic_path
				is_cultivator = yes
				is_adult = yes
				NOT = { this = root }
			}
			save_scope_as = ambusher
		}
	}

	option = {
		name = xianxia_path.1001.a
		trigger = { exists = scope:ambusher }
		duel = {
			skill = prowess
			target = scope:ambusher
			52 = {
				earn_cultivation_xp = yes
				add_trait_xp = { trait = righteous_path value = 80 }
				add_prestige = 180
				add_piety = 100
				random_list = {
					60 = { add_gold = 80 }
					40 = { add_character_modifier = { modifier = focusing_cultivation_modifier months = 6 } }
				}
				random_list = {
					40 = { add_rival = scope:ambusher }
					20 = { add_nemesis = scope:ambusher }
					40 = { }
				}
			}
			48 = {
				add_stress = 70
				remove_gold = 120
				random_list = {
					55 = { add_trait = wounded_1 }
					20 = { add_trait = wounded_2 }
					25 = { }
				}
				random_list = {
					35 = { add_rival = scope:ambusher }
					35 = { add_nemesis = scope:ambusher }
					30 = { }
				}
			}
		}
	}

	option = {
		name = xianxia_path.1001.b
		add_stress = 35
		remove_gold = 90
		if = {
			limit = { exists = scope:ambusher }
			random_list = {
				50 = { add_rival = scope:ambusher }
				15 = { add_nemesis = scope:ambusher }
				35 = { }
			}
		}
	}
}

# Demonic ruler seeks a cultivation cauldron from a rival court
xianxia_path.1101 = {
	type = character_event
	title = xianxia_path.1101.t
	desc = xianxia_path.1101.d
	theme = intrigue

	trigger = {
		has_trait = demonic_path
		is_cultivator = yes
		is_ruler = yes
		OR = { has_trait = lustful has_trait = greedy has_trait = arbitrary }
	}

	option = {
		name = xianxia_path.1101.a
		random_ruler = {
			limit = {
				NOT = { this = root }
				is_ruler = yes
				is_adult = yes
				any_courtier = {
					is_adult = yes
					is_cultivator = yes
					OR = { has_trait = yin_constitution has_trait = yang_constitution }
				}
			}
			save_scope_as = target_ruler
		}

		if = {
			limit = { exists = scope:target_ruler }
			duel = {
				skill = intrigue
				target = scope:target_ruler
				58 = {
					scope:target_ruler = {
						random_courtier = {
							limit = {
								is_adult = yes
								is_cultivator = yes
								OR = { has_trait = yin_constitution has_trait = yang_constitution }
							}
							save_scope_as = cauldron_target
						}
					}
					if = {
						limit = { exists = scope:cauldron_target }
						scope:cauldron_target = {
							if = {
								limit = { is_imprisoned = no }
								root = { imprison = { target = scope:cauldron_target type = house_arrest } }
							}
							add_character_modifier = { modifier = cultivation_furnace_target years = 8 }
						}
						add_character_modifier = { modifier = furnace_refinement_boost years = 6 }
						add_trait_xp = { trait = demonic_path value = 180 }
						add_dread = 35
					}
				}
				42 = {
					add_stress = 80
					add_rival = scope:target_ruler
					add_prestige = -80
				}
			}
		}
		else = {
			add_stress = 20
			add_prestige = -40
		}
	}

	option = {
		name = xianxia_path.1101.b
		add_stress = -15
	}
}
